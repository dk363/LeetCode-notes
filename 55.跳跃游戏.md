# 贪心
```c++
class Solution {
public:
    bool canJump(vector<int>& nums) {
        int range = 0;
        int n = nums.size();
        if (n == 1) return true; // 如果长度等于 1 起点和终点相同
        for (int i = 0; i <= range; i++) { // i <= range 很细节
            range = max(i + nums[i], range);
            if (range >= n - 1) return true; 
        }
        return false;
    }
};
```
