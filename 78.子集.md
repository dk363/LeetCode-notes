# 回溯
题目要求我们要保存所有的子集
会不会有重复的情况呢？
题目已经说明了不会有重复的元素
如果有重复的元素，像之前一样，重复元素的结果都是一样的，所以如果有重复的元素我们跳过即可
```c++
class Solution {
    vector<int> path;
    vector<vector<int>> ans;
    void dfs(const vector<int>& nums, int start) {
        ans.push_back(path);
        if (start > nums.size()) {
            return;
        }

        for (int i = start; i < nums.size(); i++) {
            // if (i > start && nums[i] == nums[i - 1]) {
            //     continue;
            // }
            path.push_back(nums[i]);
            dfs(nums, i + 1);
            path.pop_back();
        }
    }
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        // ranges::sort(nums);
        dfs(nums, 0);
        return ans;      
    }
};
```
