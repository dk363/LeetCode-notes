# 模拟
这一道题在思路上很简单
但是对于细节的处理很要紧
```cpp
class Solution {
    vector<string> convert_to_vector(const string& s) {
        vector<string> ans;
        string token;
        stringstream ss(s);

        
        while (getline(ss, token, '.')) {
            // 去掉前导 0
            int pos = 0;
            while (pos < (int)token.size() && token[pos] == '0') {
                ++pos;
            }
            string filtered = token.substr(pos);
            // 对于全部为 0 的情况
            if (filtered.empty()) filtered = "0";
            ans.push_back(filtered);
        }
        return ans;
    }
public:
    int compareVersion(string version1, string version2) {
        vector<string> v1 = convert_to_vector(version1);
        vector<string> v2 = convert_to_vector(version2);

        for (int i = 0; i < max(v1.size(), v2.size()); ++i) {
            // 对于 min(v1.size() v2.size()) 来说
            // 如果前面相等 但是一个没有一个后面还有的情况来说
            // 这里通过将其处理为 0 
            int num1 = (i < v1.size()) ? stoi(v1[i]) : 0;
            int num2 = (i < v2.size()) ? stoi(v2[i]) : 0;

            if (num1 < num2) return -1;
            if (num1 > num2) return 1;
        }
        return 0;
    }
};
```
