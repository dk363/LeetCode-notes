# 回溯 + 有效的字符判断
有效的IP地址判断
答案要求我们返回一串有效的IP地址
回溯两个点：
1. 限制条件
2. 在哪里分割

这里我们可以利用 '.' 来进行分割
通过对于有效的部分IP地址进行分割，然后维护一个 startIndex 和 endIndex 来继续递归下一个子串

```c++
class Solution {
    vector<string> ans;
    
    void dfs(string& s, int start, int pointNum) {
        if (pointNum == 3) {
            if (isValid(s, start, s.size() - 1)) {
                ans.push_back(s);
            }
            return;
        }

        for (int i = start; i < s.size(); i++) {
            if (isValid(s, start, i)) {
                s.insert(s.begin() + i + 1, '.');
                dfs(s, i + 2, pointNum + 1);
                s.erase(s.begin() + i + 1);
            }
        }
    }

    bool isValid(const string& s, int start, int end) {
        // 剪枝
        if (end < start || end - start > 2) {
            return false;
        }

        // 不能有前导 0
        if (s[start] == '0' && start != end) {
            return false;
        }

        int num = 0;
        for (int i = start; i <= end; i++) {
            if (!isdigit(s[i])) {
                return false;
            }
            num = num * 10 + (s[i] - '0');
        }
        return num < 256;
    }
public:
    vector<string> restoreIpAddresses(string s) {
        dfs(s, 0, 0);
        return ans;
    }
};
```
