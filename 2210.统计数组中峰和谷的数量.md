# 三指针 + 一次遍历
```c++
class Solution {
public:
    int countHillValley(vector<int>& nums) {
        int ans = 0;
        int pre = nums[0];
        for (int i = 1; i + 1 < nums.size(); i++) {
            int cur = nums[i];
            int nxt = nums[i + 1];

            if (cur == nxt) {
                continue;
            }

            // pre 有可能 == cur
            if (pre != cur && (pre < cur) == (cur > nxt)) {
                ans++;
            }
            // pre == cur 的时候，我们依然可以更新. 没有影响
            pre = cur;
        }
        return ans;
    }
};
```
