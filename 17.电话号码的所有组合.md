# 回溯
```c++
class Solution {
    const string MAPPING[10] = {"", "", "abc", "def", "ghi", "jkl", "mno", "pqrs", "tuv", "wxyz"};

public:
    vector<string> letterCombinations(string digits) {
        int n = digits.size();
        if (n == 0) {
            return {};
        }

        vector<string> ans;
        string path(n, 0);

        auto dfs = [&](this auto&& dfs, int i) -> void {
            if (i == n) {
                ans.push_back(path);
                return;
            }
            // 对于每一个该位置上的字母，我们都遍历 dfs
            // 这里可以直接覆写之前的字母
            // 所以这里我们不需要pop
            for (char c : MAPPING[digits[i] - '0']) {
                path[i] = c;
                dfs(i + 1);
            }
        };

        dfs(0);

        return ans;
    }
};
```
