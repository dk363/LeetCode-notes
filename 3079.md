# simulation
可以化简为 最大的数 * base 
但是是如何想到 base 的呢？
要找到最大的数字 然后用最大的数字填充每一个位子
```cpp
class Solution {
public:
    int sumOfEncryptedInt(vector<int>& nums) {
        int sum = 0;
        for (int i = 0; i < nums.size(); i++) {
            int mx = 0, base = 0;
            for ( ; nums[i] > 0; nums[i] /= 10) {
                mx = max(nums[i] % 10, mx);
                base = base * 10 + 1;
            }
            sum += (mx * base);
        }
        return sum;
    }
};
```
