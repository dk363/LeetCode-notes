# 动态规划 + 滑动窗口
dp[i] 表示 从 i 开始到达区间 [k, n] 的概率
dp[i] = (dp[i + 1] + ... + dp[i + maxPts]) / maxPts
这也是一个 定长滑窗 
```cpp
class Solution {
public:
    double new21Game(int n, int k, int maxPts) {
        // f[i] 表示从 i 开始到达 [k, n] 区间的概率
        // f[i] = (f[i + 1] + ... + f[i + maxPts]) / maxPts
        // 定长滑动窗口
        vector<double> dp(n + 1);
        double s = 0;
        for (int i = n; i >= 0; i--) {
            dp[i] = i >= k ? 1 : s / maxPts;
            s += dp[i];
            if (i + maxPts <= n) {
                // 到不了 n
                s -= dp[i + maxPts];
            }
        }
        return dp[0];
    }
};
```
