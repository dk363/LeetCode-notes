# 快慢指针 + 数学推理
设 头节点到入环处 为 a
环长为 c
快指针 走了 2b
慢指针走了 b
2b - b = kc
b = kc
b - a = kc - a
也就是说这个时候快慢指针离入环处的距离恰好为 a
那么这个时候我们让快慢指针 和头指针 同时启动
直到二者相等的时候停下
```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        ListNode* slow = head;
        ListNode* fast = head;
        while (fast && fast->next) {
            fast = fast->next->next;
            slow = slow->next;
            if (fast == slow) {
                while (slow != head) {
                    slow = slow->next;
                    head = head->next;
                }
                return slow;
            }
        }
        return nullptr;
    }
};
```
