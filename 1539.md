```cpp
class Solution {
public:
    int findKthPositive(vector<int>& arr, int k) {
        /**
            假设前面全部缺失
            然后遍历数组
            如果当前的元素是小于 k 的
            那么说明这个时候有元素 在 k 之前出现
            然后将 k++
            相当于是站好了位置    
        */
        for (int i : arr) {
            if (i <= k) {
                k++;
            } else {
                break;
            }
        }
        return k;
    }
};
```
