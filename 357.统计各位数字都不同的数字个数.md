# 排列 
给你九个数字 固定的数字长度 n 每一位的数字都不相同
```c++
class Solution {
public:
    int countNumbersWithUniqueDigits(int n) {
        if (n == 0) return 1;
        int ans = 10, uniqueDigits = 9, aviliable = 9;
        for (int i = 2; i <= n && aviliable > 0; i++) {
            uniqueDigits *= aviliable;
            ans += uniqueDigits;
            aviliable--;
        }
        return ans;
    }
};
```
# 回溯
枚举下一个是哪一个数字
用 seen 数组保存已经用过的数字
```c++
class Solution {
public:
    int countNumbersWithUniqueDigits(int n) {
        vector<uint8_t> seen(10, false);
        int ans = 0;

        auto dfs = [&](this auto&& dfs, int i, bool isStart) -> void {
            if (i > n) return;
            if (i >= 0) ans++;

            for (int j = 0; j <= 9; j++) {
                if (seen[j]) continue;
                if (j == 0 && isStart) continue;
                seen[j] = true;
                dfs(i + 1, false);
                seen[j] = false;
            }
        };

        dfs(0, true);
        return ans;
    }
};
```
