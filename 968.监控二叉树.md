# DP + 分类讨论
树节点有三种情况
1. 自己是监控
2. 父节点是监控
3. 子节点是监控
base case 是空节点，空节点只有一个参数 choose = inf
```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
    tuple<int, int, int> dfs(TreeNode* node) {
        if (node == nullptr) {
            return {INT_MAX / 2, 0, 0};
        }
        auto [l_choose, l_by_fa, l_by_children] = dfs(node->left);
        auto [r_choose, r_by_fa, r_by_children] = dfs(node->right);
        // 如果选择自己当监控 那么左子节点可以选择三种
        int choose = min({l_choose, l_by_fa, l_by_children}) + 
                     min({r_choose, r_by_fa, r_by_children}) + 1;
        // 如果父节点是监控 那么除去左子节点by_fa 的情况，因为左子节点是by_fa 就不能保证左子节点可以被监控到
        int by_fa = min(l_choose, l_by_children) + min(r_choose, r_by_children);
        // 如果子节点是监控 有三种情况 左边是 右边是 两边都是
        int by_children = min({l_choose + r_by_children, 
                               r_choose + l_by_children, 
                               r_choose + l_choose});
        return {choose, by_fa, by_children};
    }
public:
    int minCameraCover(TreeNode* root) {
        auto[choose, by_fa, by_children] = dfs(root);
        // 因为根节点没有 父节点的情况 所以肯定是 by_fa == 0 
        return min(choose, by_children);
    }
};
```
